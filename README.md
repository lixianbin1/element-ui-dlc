### 安装

```sh
yarn add element-ui-dlc  # npm install element-ui-dlc
```

### 使用

```js
import Vue from 'vue';
import ElementUiDLC from 'element-ui-dlc';
Vue.use(ElementUiDLC)
```

```html
<template>
  <div id="app">
    <el-button round>取 消</el-button>
  </div>
<template>
```
